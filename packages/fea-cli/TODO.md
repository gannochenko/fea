~~~
fea create // create a local branch
fea submit [branch_id] // push the current or specified branch
fea merge [branch_id] // merge the current or specified branch
fea info [branch_id] // get info on the current or specified branch
~~~

~~~
fea checkout [branch_id] // switch to another branch
~~~

~~~
fea release create // create a new release
fea release merge // merge current release
~~~
